sendData=function(tag,url,data,callback){
  	$.ajax({
  		url:url,
  		dataType:"json",
  		data:data,
  		tag:tag,
  		success:function(res){
  			if(res.status == 1){
  				if(tag == 0){
  					if( res.data[0].imageUrl!="" && res.data[0].url ){
  						active.NowdataEmpty = true;
  					}
  				}else if(tag == 1 ){
  					if( res.data[0].imageUrl && res.data[0].url ){
  						active.PastdataEmpty = true;
  					}
  				}
  				typeof callback == "function" && callback(res,tag);
  			}
  		}
  	})
}

	if(statu==0){

		_self.jsonNow = ret.data;
	}else if(statu==1){
		_self.jsonPast = ret.data;
	}

}
